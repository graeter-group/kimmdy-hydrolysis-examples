read_recipes = function(path) {
  recipes = read_csv(path)
  rates = recipes |>
    select(-timespans) |>
    separate_longer_delim(rates, ",") |>
    mutate(
      rate = parse_number(rates)
    )

  ts = recipes |>
    select(timespans) |>
    separate_longer_delim(timespans, "), (") |>
    separate_wider_delim(timespans, ",", names = c("from", "to")) |>
    mutate(across(c(from, to), parse_number))

  rates |>
    add_column(
      from = ts$from,
      to = ts$to
    ) |>
    mutate(
      reaction = if_else(str_detect(recipe_steps, "get_steps"), "hydrolysis", "homolysis")
    )
}

just_hydrolysis = function(recipes, bonds = tibble()) {
  recipes |>
    filter(reaction == "hydrolysis") |>
    mutate(
      key = as.integer(str_extract(recipe_steps, "<(\\d+),", group = 1)),
    ) |>
    left_join(bonds, by = "key") |>
    mutate(
      has_hyp = case_when(
        c == "HYP" & n == "HYP" ~ "HYP at C and N",
        c == "HYP" ~ "HYP at C",
        n == "HYP" ~ "HYP at N",
        TRUE ~ "no HYP"
      ),
      has_pro = case_when(
        c == "PRO" & n == "PRO" ~ "PRO at C and N",
        c == "PRO" ~ "PRO at C",
        n == "PRO" ~ "PRO at N",
        TRUE ~ "no PRO"
      )
    )
}

just_homolysis = function(recipes) {
  recipes |>
    filter(reaction == "homolysis")
}

label <- function(labels, multi_line = TRUE, sep = ": ") {
  value <- ggplot2:::label_value(labels, multi_line = multi_line)
  variable <- ggplot2:::label_variable(labels, multi_line = multi_line)
  if (multi_line) {
    out <- vector("list", length(value))
    for (i in seq_along(out)) {
      unit = ifelse(variable[[1]] == "force", "nN", "")
      sep = ifelse(variable[[1]] == "force", "", "")
      k = ifelse(variable[[1]] == "ph", variable[[i]], "")
      out[[i]] <- paste0(k, sep, " ", value[[i]], " ", unit)
    }
  } else {
    value <- inject(paste(!!!value, sep = ", "))
    variable <- inject(paste(!!!variable, sep = ", "))
    out <- Map(paste, variable, value, sep = sep)
    out <- list(unname(unlist(out)))
  }
  out
}
